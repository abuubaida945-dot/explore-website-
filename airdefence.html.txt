    <script>
        const canvas = document.getElementById('canvas');
        const ctx = canvas.getContext('2d');
        const W = canvas.width, H = canvas.height;
        let score = 0;
        const scoreSpan = document.getElementById('scoreValue');

        let enemies = [];
        let bullets = [];
        let gunX = W/2;

        function spawnEnemy() {
            enemies.push({
                x: Math.random() * (W-40) + 20,
                y: -20,
                speed: 1.5 + Math.random() * 2
            });
        }

        canvas.addEventListener('click', (e) => {
            const rect = canvas.getBoundingClientRect();
            const scaleX = W / rect.width;
            const mouseX = (e.clientX - rect.left) * scaleX;
            gunX = mouseX;
            bullets.push({ x: gunX, y: H-40, speed: 8 });
        });

        function update() {
            // Move enemies
            enemies.forEach(e => e.y += e.speed);
            // Remove off-screen enemies
            enemies = enemies.filter(e => e.y < H+20);
            // Move bullets
            bullets.forEach(b => b.y -= b.speed);
            bullets = bullets.filter(b => b.y > -20);
            // Collision detection
            for (let b of bullets) {
                for (let e of enemies) {
                    let dist = Math.hypot(b.x - e.x, b.y - e.y);
                    if (dist < 20) {
                        e.toRemove = true;
                        b.toRemove = true;
                        score++;
                    }
                }
            }
            enemies = enemies.filter(e => !e.toRemove);
            bullets = bullets.filter(b => !b.toRemove);
            scoreSpan.textContent = score;
        }

        function draw() {
            ctx.clearRect(0, 0, W, H);
            // Ground
            ctx.fillStyle = '#1a3b1a';
            ctx.fillRect(0, H-20, W, 20);
            // Gun
            ctx.fillStyle = '#888';
            ctx.fillRect(gunX-10, H-45, 20, 30);
            ctx.fillStyle = '#aaa';
            ctx.fillRect(gunX-5, H-70, 10, 30);
            // Enemies
            ctx.fillStyle = 'red';
            enemies.forEach(e => {
                ctx.beginPath();
                ctx.arc(e.x, e.y, 10, 0, 2*Math.PI);
                ctx.fill();
                // trail
                ctx.strokeStyle = 'rgba(255,0,0,0.4)';
                ctx.beginPath();
                ctx.moveTo(e.x, e.y);
                ctx.lineTo(e.x, e.y-20);
                ctx.stroke();
            });
            // Bullets
            ctx.fillStyle = 'cyan';
            bullets.forEach(b => {
                ctx.beginPath();
                ctx.arc(b.x, b.y, 5, 0, 2*Math.PI);
                ctx.fill();
            });
        }

        function loop() {
            update();
            draw();
            requestAnimationFrame(loop);
        }

        setInterval(spawnEnemy, 1500);
        loop();
    </script>
